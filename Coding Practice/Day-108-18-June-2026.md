# Rollback Simulation

## 🔹 Jenkins Pipeline - Rollback on Failure
You will learn how to **simulate rollback if deployment fails**.

pipeline {
    agent any
    environment {
        INSTITUTE_NAME = "Codr-Institute"
    }
    stages {
        stage('Deploy') {
            steps {
                script {
                    try {
                        sh 'echo "Deploying $INSTITUTE_NAME application..."'
                        sh 'exit 1' // simulate failure
                    } catch (Exception e) {
                        echo "Deployment failed. Rolling back $INSTITUTE_NAME application..."
                    }
                }
            }
        }
    }
}

## 🔹 GitLab CI - Rollback Simulation
You will learn how to **simulation rollback using after_script**.

stages:
  - deploy

variables:
  INSTITUTE_NAME: "Codr-Institute"

deploy:
  stage: deploy
  image: alpine:latest
  script:
    - echo "Deploying $INSTITUTE_NAME application..."
    - exit 1
  after_script:
    - echo "Deployment failed. Rolling back $INSTITUTE_NAME application..."

# Day 18 - High Availability, Multi-Region, and Auto-scaling

## 🔹 Ansible - Configure HAProxy for Load Balancing

```yaml
- name: Setup HAProxy Load Balancer
  hosts: all
  become: yes
  tasks:
    - name: Install HAProxy
      yum:
        name: haproxy
        state: present
    - name: Configure HAProxy
      template:
        src: haproxy.cfg.j2
        dest: /etc/haproxy/haproxy.cfg
    - name: Start HAProxy service
      service:
        name: haproxy
        state: started
        enabled: yes
🔹 Terraform - Multi-Region Auto-scaling Group with ALB 
resource "aws_lb" "codr-lb" {
   name               = "codr-lb"

🔹 Kubernetes - Horizontal Pod Autoscaler with Multiple Replicas

apiVersion: apps/v1
kind: Deployment
metadata:
  name: codr-web-app
spec:
  replicas: 3
  selector:
    matchLables:
      app: codr-web-app
  template:
    metadata:
      labels:
        app: codr-web-app
    spec:
      containers:
        - name: nginx
          image: nginx
          ports:
            - containerPort: 80
        
---
apiVersion: autoscaling/v2
kind: HorizontalPodAutoscaler
metadata:
  name: codr-web-app-hpa
spec:
  scaleTargetRef:
    apiVersion: apps/v1
    kind: Deployment
    name: codr-web-app
  minReplicas: 2
  maxReplicas: 10
  targetCPUUtilizationPercentage: 80

🔹 Jenkinsfile - Blue-Green Deployment Strategy
pipeline {
  agent any
  stages {
    stage('Build') {
       steps {
          echo 'Building Docker image...'
          docker.build('codr-web-app')
       }
    }
    stage('Test') {
       steps {
          echo 'Running tests...'
       }
    }
    stage('Blue-Green Deployment') {
       steps {
          script {
             sh 'kubectl apply -f kubernetes/blue-deployment.yaml'
             sh 'kubectl apply -f kubernetes/green-deployment.yaml'
             sh 'kubectl rollout status deployment/blue-deployment'
             sh 'kubectl rollout status deployment/green-deployment'
          }
       }
    }
  }
}
