## 1️⃣ Terraform 
- Create EC2 (choose: c5.xlarge / r5.2xlarge / r6i.4xlarge / c6i.8xlarge / c6a.12xlarge)
- Security group
- key pair
- EBS volume
- Output IP

terraform {
    required_version = ">= 1.5.0"

    required_providers {
        aws = {
            source = "hashicorp/aws"
            version = "~> 5.0"
        }
    }
}

provider "aws" {
    region = "us-east-1"
}

# Get latest Amazon Linux 2023 AMI
data "aws_ami" "amazon_linux" {
    most_recent = true

    owners = ["amazon"]
    
    filter {
        name = "name"
        values = ["al2023-ami-*-x86_64"]
    }
}

# Create Key Pair
resource "aws_key_pair" "ec2_key" {
    key_name = "terraform-key"
    public_key = file("id_rsa.pub")
}

# Security Group
resource "aws_security_group" "ec2_sg" {
    name = "terraform-ec2-sg"
    description = "Allow SSH and HTTP"

    ingress {
        description = "HTTP"
        from_port = 80
        to_port = 80
        protocol = "tcp"
        cidr_blocks = "[0.0.0.0/0"]
    }

    tags = {
        Name = "terraform-ec2-sg"
    }
}

# EC2 Instance
resource "aws_instance" "server" {
    ami = data.aws.ami.amazon_linux.id
    instance_type = "c5.xlarge" # choose from given options
    key_name = aws_key_pair.ec2_key.key_name
    vpc_security_group_ids = [aws_security_group.ec2_sg.id]

    tags = {
        Name = "terraform-server"
    }
}

# Additional EBS Volume 
resource "aws_ebs_volume" "extra_volume" {
    availability_zone = aws_instance.server.availability_zone
    size = 50
    type = "gp3"

    tags = {
        Name = "extra-ebs-volume"
    }
}

# Attach EBS Volume to EC2 
resource "aws_volume_attachment" "ebs_attach" {
    device_name = "/dev/sdf"
    volume_id = aws_ebs_volume.extra_volume.id
    instance_id = aws_instance.server.id
}

## 2️⃣ Ansible
- Install nginx
- Deploy "Welcome to Pathnex" HTML page
- Start & enable service

---
- name: Install Nginx (Debian/Ubuntu)
  apt:
    name: nginx
    state: present
    update_cache: yes
    when: ansible_os_family == "Debian"

- name: Install Nginx (RHEL/CentOS)
  yum:
    name: nginx
    state: present
  when: ansible_os_family == "RedHat"

- name: Deploy Welcome Page
  copy:
    dest: /var/www/html/index.html
    content: |
        <!DOCTYPE html>
          <html>
          <head>
              <title>Pathnex</title>
          </head>
          <body>
              <h1>Welcome to Pathnex</h1>
          </body>
          </html>
    owner: root
    group: root
    mode: '0644'

- name: Start and Enble Nginx Service
  service:
    name: nginx
    state: started
    enabled: yes


## 3️⃣ Kubernetes
- Create Deployment with PVC
- Expose via NodePort

---
apiVersion: v1
kind: PersistentVolumeClaim
metadata:
  name: nginx-pvc
spec:
  accessModes:
     - ReadWriteOnce
  resources:
     requests:
        storage: 1Gi

---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: nginx-deployment
spec:
  replicas: 1
  selector:
    matchLabels:
      app: nginx
  template:
    metadata:
      labels:
        app: nginx
    spec:
      containers:
        - name: nginx
          image: nginx:latest
          ports:
            - containerPort: 80
          volumeMounts:
            - name: nginx-storage
              mountPath: /usr/share/nginx/html
      volumes:
        - name: nginx-storage
          persistentVolumeClaim:
            claimName: nginx-pvc

---
apiVersion: v1
kind: Service
metadata: 
  name: nginx-nodeport
spec:
  type: NodePort
  selector:
    app: nginx
  ports:
    - port: 80
      targetPort: 80
      nodePort: 30080

## 4️⃣ Shell Script
- Print 5 system checks:
  - CPU
  - Memory
  - Disk
  - IP
  - Top process


#!/bin/bash

echo "System Checks"

# 1. CPU Information
echo -e "\n1. CPU Information:"
lscpu | grep "Model name"

# 2. Memory Information
echo -e "\n2. Memory Information:"
free -h

# 3. Disk Information
echo -e "\n3. Disk Information:"
df -h /

# 4. IP Address
echo  -e "\n4. IP Address:"
hostname -I

# 5. Top Process (Highest CPU Usage)
echo -e "\5. Top Process (by CPU Usage):"
ps -eo pid,come,%cpu --sort=-%cpu | head -n 2
