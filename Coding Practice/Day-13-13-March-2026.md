# Matrix Builds

## 🔹 Jenkins Pipeline - Matrix Build
you will learn how to **run the same build on multiple environments**.

pipeline {
    agent any
    environment {
        INSTITUTE_NAME = "Pathnex"
    }
    matrix {
        axes {
            axis {
            name 'JAVA_VERSION'
            values '8','11','27'
             }
        }
        stages {
            stage('Build') {
                steps {
                    sh "echo 'Building With Java $JAVA_VERSION'"
                }
            }
        }
    }
}

    server {
        listen 80;

            root /usr/share/nginx/html;
            index index.html index.htm;
        }
    }

  name: pathnex-nginx

  replicas: 2
  selector:
    matchLabels:
       app: pathnex-nginx

       labets:

         - name: nginx
           image: nginx

🔹 Shell Script - Check for Running Services 
#!/bin/bash
services=("nginx" "docker" "httpd")
for service in "${services[@]}"
do
  if systemctl is-active --quite $service; then
     echo "$service is running"
  else 
     echo "$service is not running"
  fi
done
