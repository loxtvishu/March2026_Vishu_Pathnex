    - name: Install Docker

    - name: Start Docker

🔹 Kubernetes - Pod with Volume Mount
apiVersion: v1
kind: Pod
metadata:
  name: pathnex-pod
spec:
  containers:
     - name: web
       image: nginx
       volumeMounts:
          - mountPath: /usr/share/nginx/html
            name: html-volume
  volume:
     - name: html-volume
       hostPath:
          path: /data
          type: Directory

🔹 Shell Script - Check for Running Services

#!/bin/bash
services=("nginx" "docker" "httpd")
for service in "${services[@]}"
do
  if systemctl is-active --quite $service; then
     echo "$service is running"
  else
     echo "$service is not running"
  fi
done
