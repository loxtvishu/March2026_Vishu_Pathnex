#!/bin/bash

while true
do 
   echo "============================"
   echo "      System Menu"
   echo "============================"
   echo "1) IP Address"
   echo "2) Disk Usage"
   echo "3) Memory Usage"
   echo "4) Exit"
   echo "============================"

   read -p "Enter your choice: " choice

   case $choice in
        1)
           echo "IP Address:"
           hostname -I
           ;;
        2)
           echo "Disk Usage:"
           df -h
           ;;
        3)  
           echo "Memory Usage:"
           free -h
           ;;
        4)
           echo "Exiting..."
           exit 0
           ;;
        *)
           echo "Invailed choice! Please select 1-4."
           ;;
    esac

    echo
    read -p "Press Enter to continue..."
    clear
done


## 5пёЏвѓЈ Jenkins Pipeline
You will learn to **automate all tasks in a Jenkins Pipeline** under real-world stages:

- Create **Declarative Pipeline** with stages:
- Infrastructure вЂ“ Terraform EC2 with tag Project = "Pathnex-Training"
- App Deployment вЂ“ Ansible user creation, SSH setup, Kubernetes deployment
- Monitoring Checks вЂ“ Shell menu to display:
- IP
- Disk
- Memory

- Use environment variables for all stages:
- INSTITUTE_NAME = "Pathnex"
- TEAM = "DevOps"
- ENV = "prod"
- PROJECT = "Pathnex-Training"


pipeline {
  agent any

  environment {
    INSTITUTE_NAME = "Pathnex"
    TEAM = "DevOps"
    ENV = "prod"
    PROJECT = "Pathnex-Trainig"
  }

  stages {
    stage('Infrastrusture') {
      steps {
        echo "Creating EC2 using Terraform..."

        dir('terraform') {
          sh '''
              terraform init

              terraform plan \
              -var="project=${PROJECT}" \
              -out=tfplan

              terraform apply -auto-approve tfplan
          '''
        }
      }
    }

    stage('App Deployment') {
      steps {
        echo "Running Ansible Playbook..."

        dir('ansible') {
          sh '''
              ansible-playbook -i inventory.ini user_setup.yml
          '''
        }

        echo "Deploying Kubernetes Application..."

        sh'''
           Kubectl apply -f deployment.yaml
        '''
      }
    }

    stage('Monitoring Checks') {
      steps {
        sh '''
        while true
        do
           echo "==============================="
           echo "Institute   : ${INSTITUTE_NAME}"
           echo "Team        : ${TEAM}"
           echo "Environment : ${ENV}"
           echo "Project     : ${PROJECT}"
           echo "==============================="
           echo "1) IP Address"
           echo "2) Disk Usage"
           echo "3) Memory Usage"
           echo "4) Exit"
           
           read -p "Enter Choice: " choice

           case $choice in
                1)
                   echo "IP Address:"
                   hostname -I
                   ;;
                2)
                   echo "Disk Usage:"
                   df -h
                   ;;
                3) 
                   echo "Memory Usage:"
                   free -h
                   ;;
                4)
                   echo "Exiting..."
                   break
                   ;;
                *)
                   echo "Invailed Choice!"
                   ;;
            esac

            echo ""
        done
        '''
      }
    }
  }

  post {
    sucess {
      echo "Pipeline Completed sucessfully."
    }

    failure {
      echo "Pipeline Failed."
    }
    always {
      echo "Pipeline execution finished."
    }
  }
}
