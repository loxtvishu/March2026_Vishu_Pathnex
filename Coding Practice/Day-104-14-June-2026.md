### Security Group

    name  = "ec2-security-group"
    vpc_id = aws_vpc.main.id

    ingress {
        description = "SSH Access"
        from_port = 22
        to_port = 22
        protocol = "tcp"
        cidr_blocks = ["0.0.0.0/0"]
    }

    ingress {
        description =  "HTTP Access"
        from_port = 80
        to_port = 80

        from_port = 0
        to_port = 0

        Name  = "ec2-sg"

## 2️⃣ Ansible — Install Nginx & Deploy HTML File

- name: Install Nginx
  hosts: all
  become: yes
  
  tasks:
    
     - name: Install Nginx
       yum:
         name: nginx
         state: present
    
    - name: Start Nginx Service
      service:
         name: nginx
         state: present
         enabled: yes
    
    - name: Deploy HTML Files
      copy:
         src: index.html
         dest: /usr/share/nginx/html/index.html
         owner: root
         group: root
         mode: '0644' 
    
    - name: Restart Nginx
      service:
         name: nginx
         state: restarted
