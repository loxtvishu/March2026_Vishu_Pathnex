## 🔹 Jenkins Pipeline - Email Notifications

pipeline {
    agent any
    environment {
        INSTITUTE_NAME = "Coder-Academy"
    }
    stages{
        stage('Checkout') {
            steps {
                git url: 'https://github.com/Pathnex/sample-java-app.git'
            }
        }
        stage('Build'){
            steps {
                sh 'mvn clean package'
            }
        }
    }
    post {
        sucess {
            mail to: 'team@codracademy.com', subject: "SUCESS: Build #${env.BUILD_NUMBER}",
            body: "Build completed sucessfully for $INSTITUTE_NAME"
        }
        failure {
            mail to: 'team@coderacademy.com', subject: "FAILURE: Build #${env.BUILD_NUMBER}", body: "Build failed for $INSTITUTE_NAME"
        }
    }
}

## 🔹 Ansible - Install Helm on Server
- name: Install Helm on Server with Kubernetes
  hosts: all

  tasks: 
    - name: Download Helm with Kubernetes
      get_url:
        url: https://get.helm/sh/helm-v3.7.0-linux-amd64.tar.gz
        dest: /tmp/helm.tar.gz

    - name: Extract Helm
      unarchive:
        src: /tmp/helm.tar.gz
        dest: /usr/local/bin/
        remote_src: yes
    
    - name: Set Helm Binary Permissions
      file:
        path: /usr/local/bin/helm
        mode: '0755'

## 🔹 Jenkinsfile - Multi-Stage Pipeline
pipeline {
  agent any 
  environment {
    IMAGE_NAME = 'codr-web-app'
  }
  stage('Checkout') {
    steps {
      git 'https://github.com/codr/sample-repo.git'
    }
  }
  stage('Build Docker Image') {
    steps {
      script {
        docker.build("${env.IMAGE_NAME}")
      }
    }
  }
  stage('Push Docker Image') {
    steps {
      script {
        docker.withRegistry('https://docker.io','docker-credentials') {
          docker.image("${env.IMAGE_NAME}").push('latest')
        }
      }
    }
  }
  stage('Deploy to Kubernetes') {
    steps {
      script {
        sh 'Kubectl apply -f deployment.yaml'
      }
    }
  }
}
