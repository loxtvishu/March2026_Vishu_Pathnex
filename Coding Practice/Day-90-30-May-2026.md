    echo "2) Disk Usage"
    echo "3) Memory Usage"
    echo "4) Exit"
    read -p "Choose an option: " choice

        1)
            echo "IP Address:"
            hostname -I
            ;;
        2)
            echo "Disk Usage:"
            df -h
            ;;
        3)
            echo "Memory Usage:"

            ;;
    esac

    echo
done


## 5️⃣ Jenkins Pipeline
You will learn to **automate all tasks in a Jenkins Pipeline** under real-world stages:

- Create **Declarative Pipeline** with stages:
- Infrastructure – Terraform EC2 with tag Project = "Pathnex-Training"
- App Deployment – Ansible user creation, SSH setup, Kubernetes deployment
- Monitoring Checks – Shell menu to display:
- IP
- Disk
- Memory

- Use environment variables for all stages:
- INSTITUTE_NAME = "Pathnex"
- TEAM = "DevOps"
- ENV = "prod"
- PROJECT = "Pathnex-Training"

pipeline {
  agent any 

  environment {
    INSTITUTE_NAME = "Pathnex"
    TEAM = "DevOps"
    ENV = "prod"
    PROJECT = "Pathnex-Training"
  }

  stages {
    stage('Infrastructure') {
      steps {
        sh '''
        terraform init

        terraform apply -auto-approve \
           -var="project=${PROJECT}"
        '''
      }
    }

    stage('App Deployment') {
      steps {
        echo "Deploying Application Components"

        sh '''
        ansible-playbook ansible/create-user.yml \ --extra-vars "project=${PROJECT}"

        kubectl apply -f k8s/deployment.yaml
        '''
      }
    }

    stage('Monitoring Checks') {
      steps {
        echo "Running Monitoring Checks"

        sh '''
        echo "1. IP Address"
        hostname -I

        echo ""
        echo "2. Disk Usage"
        df -h 

        echo ""
        echo "3. Memory Usage"
        free -h
        '''
      }
    }
  }

  post {
    sucess {
      echo "Pipelline completed sucessfully"
    }

    failure {
      echo "Pipeline failed"
    }

    always {
      echo "Institute: ${INSTITUTE_NAME}"
      echo "Team: ${TEAM}"
      echo "Environment: ${ENV}"
      echo "Project: ${PROJECT}"
    }
  }
}
