# Artifacts Promotion

## 🔹 Jenkins Pipeline - Promote Artifacts
you will learn how to **promote build artifacts from dev to staging**.

pipeline {
    agent any
    environment {
        INSTITUTE_NAME = "Pathnex"
    }
    stages {
        stage('Build') {
            steps {
                sh 'mvn clean package'
            }
        }
        stage('Promote'){
            steps {
                input message: "Promote artifacts to staging?"
                sh 'echo "Promoting artifacts for $INSTITUTE_NAME"
            }
        }
    }
}

## 🔹 Jenkinsfile - Deploy to Multi-Cluster Environments
pipeline {
    agent any 
    stages {
        stage ('Build') {
            steps {
                echo 'Building Docker imae...'
                docker.build('pathnex-web-app')
            }
        }
        stage('Test'){
            steps {
                echo 'Running tests...'
            }
        }
        stage('Deploy to Kubernetes') {
            steps {
                script {
                    if (env.CLUSTER == 'us-east') {
                        sh 'kubectl --context=us-east apply -f kubernetes/deployment.yaml'
                    } else {
                        sh 'kubectl --context-us-west apply -f kubernetes/deployment.yaml'
                    }
                }
            }
        }
    }
}

## 🔹 GitLab CI/CD - Multi-Region Deployment with Kubernetes
stages:
  - build
  - push
  - deploy

build:
  stage: build
  script:
    - docker build -t pathnex-web-app .

push:
  stage: push
  script:
    - docker login -u "$CI_REGISTRY_USER" -p "$CI_REGISTRY_PASSWORD"
    - docker push pathnex-web-app

deploy:
  stage: deploy
  script:
    - |
      if [ "$CI_COMMIT_REF_NAME" == "main" ]; then kubectl     --context --context=us-east apply -f kubernetes/prod-deployment.yaml
      else
        kubectl --context=us-west-apply -f kubernetes/dev-devlopment.yaml
      fi
