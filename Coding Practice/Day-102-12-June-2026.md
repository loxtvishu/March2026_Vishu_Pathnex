## 🔹 Ansible - Setup Auto - Scaling Graoup (ASG)

- name: Setup Auto Scaling Group (ASG)
  hosts: all
  become: yes
  tasks:
    - name: Create Launch Configuration
      ec2_asg:
        name: Codr-asg
        region: us-east-1
        launch_config_name: codr-launch-config
        min_size: 1
        max_size: 3
        desired_capacity: 2
        vpc_zone_identifier: subnet-12344556
