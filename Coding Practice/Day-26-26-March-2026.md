# Day 26 - Packages + Services, R53 Record, InitContainer, Status Script

## 🔹 Ansible - Install + Start Multiple Services

---
- name: Install and enable services
  hosts: all
  become: yes

  tasks:
    - name: Install services
      yum:
        name:
          - httpd
          - firewalld
        state: present
    - name: Enable and start services
      services:
        name: "{{ item }}"
        state: started
        enabled: yes
      loop:
        - httpd
        - firewalld

        INSTITUTE_NAME = "Pathnex"

🔹 Kubernetes - Network Policies for Security
apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
   name: pathnex-web-policy
spec:
   podSelector:
      matchLabels:
        app: pathnex-web-app
   policyTypes:
      - Ingress
      - Egress
   ingress:
      - from:
         - podSelector
              matchLabels:
                 app: pathnex-api
   egress:
      - to:
         - podSelector:
              matchLabels:
                 app: pathnex-database

🔹 Jenkinsfile - Secure the Pipeline with Secrets
pipeline {
     agent any 
     environment {
        AWS_ACCESS_KEY_ID = credentials('aws-access-key')
        AWS_SECRET_ACCESS_KEY = credentials('aws-secret-key')
     }
     stages {
        stage('Build') {
            steps {
                echo "Building Docker image..."
                docker.build('pathnex-web-app')
            }
        }
        stage('Deploy') {
            steps {
                echo 'Deploying to AWS EC2...'
                sh 'aws ec2 run-instance --image-id ami-0abcd1234 --count 1 --instance-type t2.micro'
            }
        }
     }
}
